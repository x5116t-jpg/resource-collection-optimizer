## 🪟 Windows用インストールスクリプト

### Windows用インストール.bat

```batch
@echo off
chcp 65001 > nul
echo.
echo ====================================
echo SuperClaude フレームワーク
echo インストーラー
echo ====================================
echo.
echo このツールは Claude Code の設定ファイルを
echo 自動的にインストールします。
echo.
pause

REM バックアップディレクトリの作成
set BACKUP_DIR=%USERPROFILE%\.claude_backup_%date:~0,4%%date:~5,2%%date:~8,2%_%time:~0,2%%time:~3,2%%time:~6,2%
set BACKUP_DIR=%BACKUP_DIR: =0%

REM 既存の設定があればバックアップ
if exist "%USERPROFILE%\.claude" (
    echo.
    echo 既存の設定が見つかりました。
    echo バックアップを作成しています...
    mkdir "%BACKUP_DIR%"
    xcopy /E /I /Y "%USERPROFILE%\.claude" "%BACKUP_DIR%"
    echo ✅ バックアップ完了: %BACKUP_DIR%
    echo.
)

REM 設定ディレクトリの作成
mkdir "%USERPROFILE%\.claude" 2>nul

REM 設定ファイルのコピー
echo 設定ファイルをインストールしています...
xcopy /Y /I "config_files\*.md" "%USERPROFILE%\.claude\"

if %errorlevel% equ 0 (
    echo.
    echo ====================================
    echo ✅ インストール完了！
    echo ====================================
    echo.
    echo 設定ファイルは以下の場所にインストールされました:
    echo %USERPROFILE%\.claude
    echo.
    echo 次のステップ:
    echo 1. Claude Code を起動してください
    echo 2. プロジェクトフォルダを開いてください
    echo 3. "docs" フォルダ内の使い方ガイドを参照してください
    echo.
) else (
    echo.
    echo ❌ エラーが発生しました。
    echo "docs\トラブルシューティング.pdf" を確認してください。
    echo.
)

pause
```

## 🍎 Mac/Linux用インストールスクリプト

### Mac用インストール.command

```bash
#!/bin/bash

echo ""
echo "===================================="
echo "SuperClaude フレームワーク"
echo "インストーラー"
echo "===================================="
echo ""
echo "このツールは Claude Code の設定ファイルを"
echo "自動的にインストールします。"
echo ""
read -p "続行するには Enter キーを押してください..."

# バックアップディレクトリの作成
BACKUP_DIR="$HOME/.claude_backup_$(date +%Y%m%d_%H%M%S)"

# 既存の設定があればバックアップ
if [ -d "$HOME/.claude" ]; then
    echo ""
    echo "既存の設定が見つかりました。"
    echo "バックアップを作成しています..."
    cp -r "$HOME/.claude" "$BACKUP_DIR"
    echo "✅ バックアップ完了: $BACKUP_DIR"
    echo ""
fi

# 設定ディレクトリの作成
mkdir -p "$HOME/.claude"

# スクリプトのディレクトリを取得
SCRIPT_DIR="$( cd "$( dirname "${BASH_SOURCE[0]}" )" && pwd )"

# 設定ファイルのコピー
echo "設定ファイルをインストールしています..."
cp "$SCRIPT_DIR/config_files/"*.md "$HOME/.claude/"

if [ $? -eq 0 ]; then
    echo ""
    echo "===================================="
    echo "✅ インストール完了！"
    echo "===================================="
    echo ""
    echo "設定ファイルは以下の場所にインストールされました:"
    echo "$HOME/.claude"
    echo ""
    echo "次のステップ:"
    echo "1. Claude Code を起動してください"
    echo "2. プロジェクトフォルダを開いてください"
    echo "3. \"docs\" フォルダ内の使い方ガイドを参照してください"
    echo ""
else
    echo ""
    echo "❌ エラーが発生しました。"
    echo "\"docs/トラブルシューティング.pdf\" を確認してください。"
    echo ""
fi

read -p "終了するには Enter キーを押してください..."
```

**実行権限付与方法（Macユーザー向け説明書に記載）:**
```bash
chmod +x "Mac用インストール.command"
```

## 🎨 GUIインストーラー版（オプション）

より親切にするなら、Python + tkinter で簡単なGUIインストーラーも作成できます：

### installer_gui.py

```python
import tkinter as tk
from tkinter import ttk, messagebox
import shutil
import os
from pathlib import Path
from datetime import datetime

class SuperClaudeInstaller:
    def __init__(self, root):
        self.root = root
        self.root.title("SuperClaude フレームワーク インストーラー")
        self.root.geometry("500x400")

        # ヘッダー
        header = tk.Label(
            root,
            text="SuperClaude フレームワーク",
            font=("Arial", 16, "bold")
        )
        header.pack(pady=20)

        # 説明
        desc = tk.Label(
            root,
            text="Claude Code の設定ファイルを自動的にインストールします。\n既存の設定は自動的にバックアップされます。",
            justify=tk.LEFT
        )
        desc.pack(pady=10)

        # インストール先表示
        self.install_path = Path.home() / ".claude"
        path_frame = tk.Frame(root)
        path_frame.pack(pady=10, padx=20, fill=tk.X)

        tk.Label(path_frame, text="インストール先:").pack(anchor=tk.W)
        tk.Label(
            path_frame,
            text=str(self.install_path),
            fg="blue"
        ).pack(anchor=tk.W)

        # プログレスバー
        self.progress = ttk.Progressbar(
            root,
            mode='determinate',
            length=400
        )
        self.progress.pack(pady=20)

        # ステータスラベル
        self.status_label = tk.Label(root, text="")
        self.status_label.pack()

        # ボタン
        button_frame = tk.Frame(root)
        button_frame.pack(pady=20)

        self.install_btn = tk.Button(
            button_frame,
            text="インストール開始",
            command=self.install,
            width=15,
            bg="#4CAF50",
            fg="white"
        )
        self.install_btn.pack(side=tk.LEFT, padx=5)

        tk.Button(
            button_frame,
            text="キャンセル",
            command=root.quit,
            width=15
        ).pack(side=tk.LEFT, padx=5)

    def install(self):
        try:
            self.install_btn.config(state=tk.DISABLED)
            self.progress['value'] = 0

            # バックアップ
            if self.install_path.exists():
                self.status_label.config(text="既存設定をバックアップ中...")
                self.root.update()

                backup_dir = Path.home() / f".claude_backup_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
                shutil.copytree(self.install_path, backup_dir)
                self.progress['value'] = 30

            # ディレクトリ作成
            self.status_label.config(text="インストール準備中...")
            self.root.update()
            self.install_path.mkdir(exist_ok=True)
            self.progress['value'] = 50

            # ファイルコピー
            self.status_label.config(text="設定ファイルをコピー中...")
            self.root.update()

            config_source = Path("config_files")
            for file in config_source.glob("*.md"):
                shutil.copy(file, self.install_path / file.name)

            self.progress['value'] = 100
            self.status_label.config(text="完了！")

            messagebox.showinfo(
                "成功",
                f"インストールが完了しました！\n\n"
                f"設定ファイルは以下にインストールされました:\n"
                f"{self.install_path}\n\n"
                f"次のステップ:\n"
                f"1. Claude Code を起動\n"
                f"2. プロジェクトフォルダを開く\n"
                f"3. 使い方ガイドを確認"
            )

            self.root.quit()

        except Exception as e:
            messagebox.showerror(
                "エラー",
                f"インストール中にエラーが発生しました:\n{str(e)}"
            )
            self.install_btn.config(state=tk.NORMAL)
            self.progress['value'] = 0
            self.status_label.config(text="")

if __name__ == "__main__":
    root = tk.Tk()
    app = SuperClaudeInstaller(root)
    root.mainloop()
```
